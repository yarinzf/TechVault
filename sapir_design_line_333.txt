**toggleSMSLogin() JS (Sapir's exact logic):**
```js
function toggleSMSLogin() {
  const emailFields = document.getElementById('emailFields');
  const smsFields = document.getElementById('smsFields');
  const btnText = document.getElementById('loginBtnText');
  const smsBtn = document.getElementById('smsBtn');
  const isSMS = smsFields.style.display === 'none';
  
  emailFields.style.transition = 'opacity 0.2s ease';
  emailFields.style.opacity = '0';
  
  setTimeout(() => {
    emailFields.style.display = isSMS ? 'none' : 'block';
    smsFields.style.display = isSMS ? 'block' : 'none';
    btnText.textContent = isSMS ? 'שלח קוד אימות' : 'כניסה לחשבון';
    smsBtn.style.borderColor = isSMS ? 'var(--blue)' : '';
    smsBtn.style.color = isSMS ? 'var(--blue)' : '';
    
    const target = isSMS ? smsFields : emailFields;
    target.style.opacity = '0';
    target.style.transition = 'opacity 0.25s ease';
    requestAnimationFrame(() => { target.style.opacity = '1'; });
    lucide.createIcons();
  }, 200);
}
```
